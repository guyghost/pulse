// src/lib/core/connectors/malt-parser.ts
import type { Mission, MissionSource } from '../types/mission';

const SOURCE: MissionSource = 'malt';
const BASE_URL = 'https://www.malt.fr';

export function parseMaltHTML(html: string, now: Date, idPrefix: string): Mission[] {
  if (!html.trim()) return [];

  const parser = new DOMParser();
  const doc = parser.parseFromString(html, 'text/html');
  const missions: Mission[] = [];

  const cards = doc.querySelectorAll('.listing-card, [data-testid="freelance-listing"], .search-result');

  cards.forEach((card, index) => {
    const titleEl = card.querySelector('.listing-card__title a, h3 a, a[data-testid="listing-title"]');
    const title = titleEl?.textContent?.trim() ?? '';
    const href = titleEl?.getAttribute('href') ?? '';
    const url = href.startsWith('http') ? href : `${BASE_URL}${href}`;

    if (!title) return;

    const clientEl = card.querySelector('.listing-card__company, .company-name, [data-testid="company"]');
    const client = clientEl?.textContent?.trim() ?? null;

    const stackEls = card.querySelectorAll('.skill-badge, .listing-card__skill, [data-testid="skill"]');
    const stack = Array.from(stackEls).map(el => el.textContent?.trim() ?? '').filter(Boolean);

    const rateEl = card.querySelector('.listing-card__rate, .daily-rate, [data-testid="rate"]');
    const rateText = rateEl?.textContent?.trim() ?? '';
    const tjmMatch = rateText.match(/(\d+)/);
    const tjm = tjmMatch ? parseInt(tjmMatch[1], 10) : null;

    const locationEl = card.querySelector('.listing-card__location, .location, [data-testid="location"]');
    const location = locationEl?.textContent?.trim() ?? null;

    const durationEl = card.querySelector('.listing-card__duration, .duration, [data-testid="duration"]');
    const duration = durationEl?.textContent?.trim() ?? null;

    const descEl = card.querySelector('.listing-card__description, .description, p');
    const description = descEl?.textContent?.trim() ?? '';

    const fullText = card.textContent?.toLowerCase() ?? '';
    const remote = fullText.includes('full remote') || fullText.includes('teletravail complet') || fullText.includes('télétravail complet')
      ? 'full' as const
      : fullText.includes('hybride') || fullText.includes('hybrid')
      ? 'hybrid' as const
      : fullText.includes('sur site') || fullText.includes('on-site') || fullText.includes('onsite')
      ? 'onsite' as const
      : null;

    missions.push({
      id: `${idPrefix}-${index}`,
      title,
      client,
      description,
      stack,
      tjm,
      location,
      remote,
      duration,
      url,
      source: SOURCE,
      scrapedAt: now,
      score: null,
    });
  });

  return missions;
}

import { BaseConnector } from './base.connector';
import type { Mission } from '../../core/types/mission';
import { parseMaltHTML } from '../../core/connectors/malt-parser';
import { sendMessage } from '../messaging/bridge';

const BASE_URL = 'https://www.malt.fr';
const MISSIONS_URL = `${BASE_URL}/s?q=`;

export class MaltConnector extends BaseConnector {
  readonly id = 'malt';
  readonly name = 'Malt';
  readonly baseUrl = BASE_URL;
  readonly icon = 'user-check';

  async fetchMissions(): Promise<Mission[]> {
    const response = await sendMessage({
      type: 'SCRAPE_URL',
      payload: { url: MISSIONS_URL, connectorId: this.id },
    });

    if (response.type === 'SCRAPE_RESULT' && 'html' in response.payload) {
      const now = new Date();
      const idPrefix = `malt-${now.getTime()}`;
      const missions = parseMaltHTML((response.payload as { html: string }).html, now, idPrefix);
      await this.setLastSync();
      return missions;
    }

    return [];
  }
}

// src/lib/core/tjm/build-analysis.ts
import type { TJMAnalysis, TJMRange, TJMTrend } from '../types/tjm';
import type { AggregatedTJM } from './aggregator';

function buildRange(median: number, min: number, max: number): TJMRange {
  return {
    min: Math.round(min),
    median: Math.round(median),
    max: Math.round(max),
  };
}

function computeConfidence(count: number, stddev: number, median: number): number {
  const countFactor = Math.min(count / 20, 1);
  const cvFactor = median > 0 ? Math.max(0, 1 - (stddev / median)) : 0;
  return Math.round(countFactor * 0.6 + cvFactor * 0.4 * 100) / 100;
}

function determineTrend(stddev: number, median: number): TJMTrend {
  const cv = median > 0 ? stddev / median : 0;
  if (cv < 0.1) return 'stable';
  return 'up'; // default optimistic without historical data
}

export function buildAnalysisFromAggregation(aggregated: AggregatedTJM, now: Date): TJMAnalysis {
  const { median, min, max, count, stddev } = aggregated;

  const juniorFactor = 0.8;
  const seniorFactor = 1.25;

  const junior = buildRange(
    median * juniorFactor,
    min * juniorFactor,
    max * juniorFactor,
  );
  const confirmed = buildRange(median, min, max);
  const senior = buildRange(
    median * seniorFactor,
    min * seniorFactor,
    max * seniorFactor,
  );

  const confidence = computeConfidence(count, stddev, median);
  const trend = determineTrend(stddev, median);

  const trendDetail = trend === 'stable'
    ? `Les TJM pour ce profil sont stables autour de ${Math.round(median)} EUR/jour sur les 30 derniers jours.`
    : trend === 'up'
    ? `Tendance haussiere detectee. Le TJM median est de ${Math.round(median)} EUR/jour avec une dispersion de ${Math.round(stddev)} EUR.`
    : `Tendance baissiere. Le TJM median est de ${Math.round(median)} EUR/jour.`;

  const recommendation = confidence >= 0.6
    ? `Basee sur ${count} missions, la fourchette recommandee pour un profil confirme est ${confirmed.min}-${confirmed.max} EUR/jour.`
    : `Donnees insuffisantes (${count} mission${count > 1 ? 's' : ''}). Elargissez votre recherche pour une analyse plus fiable.`;

  return {
    junior,
    confirmed,
    senior,
    trend,
    trendDetail,
    recommendation,
    confidence,
    dataPoints: count,
    analyzedAt: now,
  };
}

  $effect(() => {

    const query = snapshot.context.query;
    if (!query) return;

    (async () => {
      try {
        const missions = await getMissions();
        const now = new Date();

        // Convert missions to TJMDataPoints
        const points: TJMDataPoint[] = missions
          .filter(m => m.tjm !== null)
          .map(m => ({
            tjm: m.tjm!,
            title: m.title,
            location: m.location,
            source: m.source,
            date: m.scrapedAt,
          }));

        const aggregated = aggregateFromPoints(points, query.title, query.location, now);

        if (!aggregated) {
          tjmActor.send({ type: 'ERROR', error: 'Pas assez de donnees. Lancez un scan depuis le feed.' });
          return;
        }

        tjmActor.send({ type: 'AGGREGATION_DONE', data: points });

        const analysisResult = buildAnalysisFromAggregation(aggregated, now);
        tjmActor.send({ type: 'LLM_DONE', analysis: analysisResult });
      } catch (err) {
        tjmActor.send({ type: 'ERROR', error: err instanceof Error ? err.message : 'Erreur inconnue' });
      }
    })();
  });

    if (!title.trim()) return;
    tjmActor.send({ type: 'ANALYZE', title, location, seniority });

Add a reactive toast list and listen for bridge error messages. Add to `App.svelte`:

```typescript
let toasts = $state<Array<{ id: number; message: string; type: 'info' | 'error' | 'success' }>>([]);

function addToast(message: string, type: 'info' | 'error' | 'success' = 'info') {
  const id = Date.now();
  toasts = [...toasts, { id, message, type }];
  setTimeout(() => { toasts = toasts.filter(t => t.id !== id); }, 4000);
}

function dismissToast(id: number) {
  toasts = toasts.filter(t => t.id !== id);
}
```

Add toast display at the bottom of the root div, before the DevPanel:
```svelte
{#if toasts.length > 0}
  <div class="absolute bottom-16 left-3 right-3 z-50 space-y-2">
    {#each toasts as toast (toast.id)}
      <Toast message={toast.message} type={toast.type} onDismiss={() => dismissToast(toast.id)} />
    {/each}
  </div>
{/if}
```
